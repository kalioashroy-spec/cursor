! -------------------------------------------------------------------------
! SUB ycs333
! 辅助子程序：
! - 结合光线追迹结果，用近似二分搜索方式求解 FLD2，
!   使 S = RAYY(NSURFACE) 在给定阈值内逼近目标 F2。
! -------------------------------------------------------------------------
SUB ycs333

declare FLD1,double,1,20
declare FLD2,double,1,20
NSURFACE = NSUR()
for i,1,NSURFACE,1
vec1(i)=SPRO(i,6)
next

for i,1,NSURFACE,1
surp i,6,vec1(i)
next


FLD1(1)=0*F1
FLD1(2)=0.3*F1
FLD1(3)=0.5*F1
FLD1(4)=0.7*F1
FLD1(5)=0.85*F1
FLD1(6)=0.9*F1
FLD1(7)=1*F1

FLD2(1)=0


FOR I,2,7,1
F2=FLD1(I)

MF=MAXF()
SYSP 101,3
FLDY(1)=0
FLDY(2)=0.5*MF
FLDY(3)=MF
SETVIG
update

MAXX=MF
MINN=0
FOR j,1,25,1
L=(MAXX+MINN)/2
P=L/MF
RAYTRACE 0,P,0,0,PWAV()
S=RAYY(NSURFACE)
E=S-F2
IF E>0
MAXX=L
else
MINN=L
ENDIF
IF (E<0.0005)&(E>-0.0005)
FLD2(I)=l
GOTO 333
ENDIF
NEXT
LABEL 333
NEXT
SYSP 101,7

FOR J,1,7,1
FLDY(J)=FLD2(J)
NEXT
SETVIG
UPDATE
RETURN
